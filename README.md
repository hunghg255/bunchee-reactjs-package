## Install

```
npm i react-marques@latest
```

## Use

```tsx
import { ReactMarques } from 'react-marques';

const App = () => {
  return (
    <ReactMarques fade={true}>
      <div>Content 1</div>
      <div>Content 2</div>
      <div>Content 3</div>
    </ReactMarques>
  );
};
```
